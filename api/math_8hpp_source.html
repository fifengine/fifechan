392 template <typename T>
393 inline T Math<T>::ACos(T _val)
394 {
395 if (-static_cast<T>(1) < _val) {
396 if (_val < static_cast<T>(1)) {
397 return static_cast<T>(std::acos(_val));
398 }
399
400 return static_cast<T>(0);
401 }
402
403 return pi();
404 }
